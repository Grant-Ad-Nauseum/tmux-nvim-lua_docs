ğŸ¯ METATABLES (ADVANCED)
-------------------------
mt = {
  __add = function(a, b)
    return {value = a.value + b.value}
  end
}

a = {value = 10}
b = {value = 20}
setmetatable(a, mt)
setmetatable(b, mt)

result = a + b
print(result.value)  -- 30
